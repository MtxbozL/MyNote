这份脚本可以实时监控 Nginx 运行状态、负载均衡后端服务器健康情况，并在出现异常时通过邮件 / 钉钉告警，可直接部署到生产环境使用。

```bash
#!/bin/bash
# Nginx 监控与告警脚本
# 功能：监控Nginx进程、端口、后端服务器健康状态，异常时发送告警

# -------------------------- 配置项（根据实际环境修改） --------------------------
# 1. 基础配置
NGINX_PID="/var/run/nginx.pid"          # Nginx PID文件路径
NGINX_CMD="/usr/sbin/nginx"             # Nginx 命令路径
NGINX_CONF="/etc/nginx/nginx.conf"      # Nginx 配置文件路径
MONITOR_PORT="80"                       # 监控的Nginx端口
UPSTREAM_SERVERS=("127.0.0.1:8080" "127.0.0.1:8081" "127.0.0.1:8082")  # 后端服务器列表
CHECK_INTERVAL=30                       # 检查间隔（秒）
LOG_FILE="/var/log/nginx/monitor.log"   # 监控日志文件

# 2. 告警配置
ALERT_TYPE="email"                      # 告警方式：email/dingding（二选一）
# 邮件告警配置
SMTP_SERVER="smtp.163.com"              # SMTP服务器
SMTP_PORT="25"                          # SMTP端口
SMTP_USER="your_email@163.com"          # 发件人邮箱
SMTP_PASS="your_email_password"         # 邮箱授权码
TO_EMAIL="alert@your-domain.com"        # 收件人邮箱
# 钉钉告警配置
DINGDING_WEBHOOK="https://oapi.dingtalk.com/robot/send?access_token=your_token"

# -------------------------- 工具函数 --------------------------
# 日志函数
log() {
    local LEVEL=$1
    local MSG=$2
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] [${LEVEL}] ${MSG}" >> ${LOG_FILE}
}

# 邮件告警函数
send_email_alert() {
    local SUBJECT=$1
    local CONTENT=$2
    log "INFO" "发送邮件告警：${SUBJECT}"
    # 使用sendmail或ssmtp发送邮件（需提前安装）
    echo -e "From: ${SMTP_USER}\nTo: ${TO_EMAIL}\nSubject: ${SUBJECT}\n\n${CONTENT}" | \
    sendmail -t
}

# 钉钉告警函数
send_dingding_alert() {
    local CONTENT=$1
    log "INFO" "发送钉钉告警：${CONTENT}"
    curl -s -X POST ${DINGDING_WEBHOOK} \
    -H "Content-Type: application/json" \
    -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"【Nginx监控告警】${CONTENT}\"}}"
}

# 统一告警函数
send_alert() {
    local SUBJECT=$1
    local CONTENT=$2
    if [ "${ALERT_TYPE}" = "email" ]; then
        send_email_alert "${SUBJECT}" "${CONTENT}"
    elif [ "${ALERT_TYPE}" = "dingding" ]; then
        send_dingding_alert "${CONTENT}"
    fi
    log "ERROR" "告警内容：${CONTENT}"
}

# -------------------------- 监控逻辑 --------------------------
# 1. 检查Nginx进程是否运行
check_nginx_process() {
    if [ ! -f ${NGINX_PID} ] || ! ps -p $(cat ${NGINX_PID}) > /dev/null 2>&1; then
        local MSG="Nginx进程已停止！PID文件：${NGINX_PID}"
        send_alert "【紧急】Nginx进程异常" "${MSG}"
        # 尝试重启Nginx
        log "INFO" "尝试重启Nginx..."
        ${NGINX_CMD} -t && ${NGINX_CMD} -s reload || ${NGINX_CMD}
        if [ $? -ne 0 ]; then
            send_alert "【严重】Nginx重启失败" "Nginx进程停止，且重启失败！请手动检查配置文件：${NGINX_CONF}"
        fi
    else
        log "INFO" "Nginx进程运行正常"
    fi
}

# 2. 检查Nginx端口是否监听
check_nginx_port() {
    if ! netstat -tulpn | grep -E ":${MONITOR_PORT}\s+LISTEN\s+.*nginx" > /dev/null 2>&1; then
        local MSG="Nginx未监听端口${MONITOR_PORT}！"
        send_alert "【紧急】Nginx端口异常" "${MSG}"
    else
        log "INFO" "Nginx端口${MONITOR_PORT}监听正常"
    fi
}

# 3. 检查后端服务器健康状态
check_upstream_servers() {
    local UNHEALTHY_SERVERS=()
    for SERVER in "${UPSTREAM_SERVERS[@]}"; do
        IP=$(echo ${SERVER} | cut -d':' -f1)
        PORT=$(echo ${SERVER} | cut -d':' -f2)
        # 使用telnet检查端口连通性（超时2秒）
        if ! timeout 2 bash -c "echo > /dev/tcp/${IP}/${PORT}" > /dev/null 2>&1; then
            UNHEALTHY_SERVERS+=("${SERVER}")
        fi
    done
    if [ ${#UNHEALTHY_SERVERS[@]} -gt 0 ]; then
        local MSG="后端服务器异常：${UNHEALTHY_SERVERS[*]}，请检查服务状态！"
        send_alert "【警告】Nginx后端服务器异常" "${MSG}"
    else
        log "INFO" "所有后端服务器健康状态正常"
    fi
}

# 4. 检查Nginx配置语法
check_nginx_config() {
    if ! ${NGINX_CMD} -t > /dev/null 2>&1; then
        local MSG="Nginx配置文件语法错误！配置文件：${NGINX_CONF}"
        send_alert "【严重】Nginx配置异常" "${MSG}"
    else
        log "INFO" "Nginx配置文件语法正常"
    fi
}

# -------------------------- 主程序 --------------------------
main() {
    # 创建日志文件（如果不存在）
    [ ! -f ${LOG_FILE} ] && touch ${LOG_FILE}
    
    log "INFO" "开始Nginx监控检查"
    check_nginx_process
    check_nginx_port
    check_nginx_config
    check_upstream_servers
    log "INFO" "Nginx监控检查完成\n"
}

# 执行主程序
main

# 可选：设置循环监控（注释掉则单次执行）
# while true; do
#     main
#     sleep ${CHECK_INTERVAL}
# done
```

### 部署与使用说明

#### 1. 环境准备

```bash
# 安装依赖工具（CentOS/RHEL）
yum install -y net-tools sendmail ssmtp curl

# 安装依赖工具（Ubuntu/Debian）
apt-get install -y net-tools sendmail curl
```

#### 2. 脚本部署

```bash
# 1. 创建脚本文件
touch /usr/local/bin/nginx_monitor.sh
chmod +x /usr/local/bin/nginx_monitor.sh

# 2. 将上述脚本内容复制到文件中
# 3. 修改脚本中的配置项（告警方式、邮箱/钉钉信息、后端服务器列表等）
vi /usr/local/bin/nginx_monitor.sh

# 4. 测试脚本
/usr/local/bin/nginx_monitor.sh

# 5. 查看监控日志
tail -f /var/log/nginx/monitor.log
```

#### 3. 设置定时任务（推荐）

```bash
# 编辑crontab
crontab -e

# 添加以下内容（每5分钟执行一次监控）
*/5 * * * * /usr/local/bin/nginx_monitor.sh > /dev/null 2>&1
```

#### 4. 钉钉告警配置（可选）

1. 登录钉钉，创建自定义机器人，获取 webhook 地址；
2. 将 webhook 地址填入脚本中的 `DINGDING_WEBHOOK`；
3. 测试告警：手动停止一个后端服务，执行脚本验证告警是否发送。

### 脚本功能说明

|监控项|检测逻辑|异常处理|
|---|---|---|
|Nginx 进程|检查 PID 文件 + 进程是否存在|发送告警 + 尝试自动重启|
|Nginx 端口|检查指定端口是否处于 LISTEN 状态|发送告警|
|后端服务器|检查后端 IP:PORT 连通性（超时 2 秒）|发送异常服务器列表告警|
|配置文件|执行`nginx -t`检查语法|发送配置错误告警|

### 总结

1. 这份脚本实现了**Nginx 核心状态监控 + 异常自动告警**，支持邮件 / 钉钉两种告警方式，可直接适配之前的生产配置模板；
2. 核心配置项需根据你的实际环境修改：后端服务器列表、告警方式、邮箱 / 钉钉信息；
3. 建议通过 crontab 设置定时执行（如每 5 分钟），确保实时监控；
4. 脚本自带日志功能，可通过`/var/log/nginx/monitor.log`追溯监控历史和告警记录。
