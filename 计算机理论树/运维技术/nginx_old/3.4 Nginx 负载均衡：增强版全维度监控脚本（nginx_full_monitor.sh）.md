
```bash
#!/bin/bash
# Nginx 全维度监控脚本
# 功能：进程/端口/后端/配置监控 + 连接数/请求量/CPU/内存监控 + 数据可视化报表

# -------------------------- 基础配置（根据实际环境修改） --------------------------
# 1. 核心配置
NGINX_PID="/var/run/nginx.pid"
NGINX_CMD="/usr/sbin/nginx"
NGINX_CONF="/etc/nginx/nginx.conf"
MONITOR_PORT="80"
UPSTREAM_SERVERS=("127.0.0.1:8080" "127.0.0.1:8081" "127.0.0.1:8082")
CHECK_INTERVAL=30
LOG_FILE="/var/log/nginx/monitor.log"
DATA_FILE="/var/log/nginx/monitor_data.csv"  # 监控数据存储文件
REPORT_FILE="/var/www/html/nginx_monitor.html"  # 可视化报表路径

# 2. 告警配置
ALERT_TYPE="dingding"  # email/dingding
# 邮件配置
SMTP_SERVER="smtp.163.com"
SMTP_PORT="25"
SMTP_USER="your_email@163.com"
SMTP_PASS="your_email_auth_code"
TO_EMAIL="alert@your-domain.com"
# 钉钉配置
DINGDING_WEBHOOK="https://oapi.dingtalk.com/robot/send?access_token=your_token"

# 3. 阈值配置（超过则告警）
CPU_THRESHOLD=80       # CPU使用率阈值（%）
MEM_THRESHOLD=80       # 内存使用率阈值（%）
CONN_THRESHOLD=2000    # Nginx连接数阈值
REQ_THRESHOLD=1000     # 每分钟请求量阈值

# -------------------------- 工具函数 --------------------------
# 日志函数
log() {
    local LEVEL=$1
    local MSG=$2
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] [${LEVEL}] ${MSG}" >> ${LOG_FILE}
}

# 邮件告警
send_email_alert() {
    local SUBJECT=$1
    local CONTENT=$2
    log "INFO" "发送邮件告警：${SUBJECT}"
    echo -e "From: ${SMTP_USER}\nTo: ${TO_EMAIL}\nSubject: ${SUBJECT}\n\n${CONTENT}" | sendmail -t
}

# 钉钉告警
send_dingding_alert() {
    local CONTENT=$1
    log "INFO" "发送钉钉告警：${CONTENT}"
    curl -s -X POST ${DINGDING_WEBHOOK} \
    -H "Content-Type: application/json" \
    -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"【Nginx全维度监控】${CONTENT}\"}}"
}

# 统一告警
send_alert() {
    local SUBJECT=$1
    local CONTENT=$2
    if [ "${ALERT_TYPE}" = "email" ]; then
        send_email_alert "${SUBJECT}" "${CONTENT}"
    elif [ "${ALERT_TYPE}" = "dingding" ]; then
        send_dingding_alert "${CONTENT}"
    fi
    log "ERROR" "告警：${CONTENT}"
}

# -------------------------- 基础监控（原有功能） --------------------------
# 检查Nginx进程
check_nginx_process() {
    if [ ! -f ${NGINX_PID} ] || ! ps -p $(cat ${NGINX_PID}) > /dev/null 2>&1; then
        local MSG="Nginx进程已停止！尝试重启..."
        send_alert "【紧急】Nginx进程异常" "${MSG}"
        ${NGINX_CMD} -t && ${NGINX_CMD} -s reload || ${NGINX_CMD}
        [ $? -ne 0 ] && send_alert "【严重】Nginx重启失败" "配置文件：${NGINX_CONF}"
    else
        log "INFO" "Nginx进程运行正常"
    fi
}

# 检查Nginx端口
check_nginx_port() {
    if ! netstat -tulpn | grep -E ":${MONITOR_PORT}\s+LISTEN\s+.*nginx" > /dev/null 2>&1; then
        send_alert "【紧急】Nginx端口异常" "未监听端口${MONITOR_PORT}！"
    else
        log "INFO" "Nginx端口${MONITOR_PORT}监听正常"
    fi
}

# 检查后端服务器
check_upstream_servers() {
    local UNHEALTHY=()
    for SERVER in "${UPSTREAM_SERVERS[@]}"; do
        IP=$(echo ${SERVER} | cut -d':' -f1)
        PORT=$(echo ${SERVER} | cut -d':' -f2)
        if ! timeout 2 bash -c "echo > /dev/tcp/${IP}/${PORT}" > /dev/null 2>&1; then
            UNHEALTHY+=("${SERVER}")
        fi
    done
    [ ${#UNHEALTHY[@]} -gt 0 ] && send_alert "【警告】后端服务器异常" "异常节点：${UNHEALTHY[*]}"
}

# 检查配置语法
check_nginx_config() {
    if ! ${NGINX_CMD} -t > /dev/null 2>&1; then
        send_alert "【严重】Nginx配置错误" "配置文件：${NGINX_CONF}"
    fi
}

# -------------------------- 新增：系统资源监控 --------------------------
# 监控CPU使用率
check_cpu_usage() {
    # 获取CPU整体使用率（排除idle）
    CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | sed "s/.*, *\([0-9.]*\)%* id.*/\1/" | awk '{print 100 - $1}')
    CPU_USAGE=$(echo ${CPU_USAGE} | cut -d'.' -f1)  # 取整数
    if [ ${CPU_USAGE} -ge ${CPU_THRESHOLD} ]; then
        send_alert "【警告】CPU使用率过高" "当前CPU使用率：${CPU_USAGE}%，阈值：${CPU_THRESHOLD}%"
    fi
    log "INFO" "CPU使用率：${CPU_USAGE}%"
    echo -n "${CPU_USAGE}," >> ${DATA_FILE}
}

# 监控内存使用率
check_mem_usage() {
    # 计算内存使用率（百分比，整数）
    MEM_TOTAL=$(free -m | grep Mem | awk '{print $2}')
    MEM_USED=$(free -m | grep Mem | awk '{print $3}')
    MEM_USAGE=$((MEM_USED * 100 / MEM_TOTAL))
    if [ ${MEM_USAGE} -ge ${MEM_THRESHOLD} ]; then
        send_alert "【警告】内存使用率过高" "当前内存使用率：${MEM_USAGE}%，阈值：${MEM_THRESHOLD}%"
    fi
    log "INFO" "内存使用率：${MEM_USAGE}%"
    echo -n "${MEM_USAGE}," >> ${DATA_FILE}
}

# -------------------------- 新增：Nginx性能监控 --------------------------
# 监控Nginx连接数
check_nginx_connections() {
    # 统计Nginx总连接数（ESTABLISHED + LISTEN）
    CONN_COUNT=$(netstat -an | grep -E ":${MONITOR_PORT}|nginx" | grep -c ESTABLISHED)
    if [ ${CONN_COUNT} -ge ${CONN_THRESHOLD} ]; then
        send_alert "【警告】Nginx连接数过高" "当前连接数：${CONN_COUNT}，阈值：${CONN_THRESHOLD}"
    fi
    log "INFO" "Nginx活跃连接数：${CONN_COUNT}"
    echo -n "${CONN_COUNT}," >> ${DATA_FILE}
}

# 监控Nginx请求量（每分钟）
check_nginx_requests() {
    # 统计最近1分钟的请求数（基于access.log）
    LOG_PATH="/var/log/nginx/access.log"
    # 获取1分钟前的时间戳
    START_TIME=$(date -d "1 minute ago" +"%d/%b/%Y:%H:%M")
    # 过滤最近1分钟的请求并计数
    REQ_COUNT=$(grep "${START_TIME}" ${LOG_PATH} | wc -l)
    if [ ${REQ_COUNT} -ge ${REQ_THRESHOLD} ]; then
        send_alert "【警告】Nginx请求量过高" "最近1分钟请求数：${REQ_COUNT}，阈值：${REQ_THRESHOLD}"
    fi
    log "INFO" "Nginx最近1分钟请求数：${REQ_COUNT}"
    echo "${REQ_COUNT}" >> ${DATA_FILE}
}

# -------------------------- 新增：可视化报表生成 --------------------------
generate_report() {
    # 创建CSV文件头（首次执行时）
    [ ! -f ${DATA_FILE} ] && echo "时间,CPU使用率(%),内存使用率(%),连接数,请求数" > ${DATA_FILE}
    
    # 写入本次监控数据（时间戳开头）
    echo -n "$(date +'%Y-%m-%d %H:%M:%S')," > temp_data.tmp
    check_cpu_usage
    check_mem_usage
    check_nginx_connections
    check_nginx_requests
    cat temp_data.tmp >> ${DATA_FILE}
    rm -f temp_data.tmp

    # 生成HTML可视化报表（使用Chart.js绘制图表）
    cat > ${REPORT_FILE} << EOF
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>Nginx监控报表</title>
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
    <style>
        body { font-family: Arial, sans-serif; margin: 20px; }
        .chart-container { width: 80%; height: 400px; margin: 20px auto; }
        h1 { text-align: center; color: #333; }
        .stats { text-align: center; font-size: 18px; margin: 10px 0; }
    </style>
</head>
<body>
    <h1>Nginx全维度监控报表</h1>
    <div class="stats">
        最后更新时间：$(date +'%Y-%m-%d %H:%M:%S') | 
        CPU使用率：$(tail -1 ${DATA_FILE} | cut -d',' -f2)% | 
        内存使用率：$(tail -1 ${DATA_FILE} | cut -d',' -f3)% | 
        Nginx连接数：$(tail -1 ${DATA_FILE} | cut -d',' -f4)
    </div>

    <!-- CPU/内存使用率图表 -->
    <div class="chart-container">
        <canvas id="resourceChart"></canvas>
    </div>

    <!-- Nginx连接数/请求量图表 -->
    <div class="chart-container">
        <canvas id="nginxChart"></canvas>
    </div>

    <script>
        // 读取CSV数据
        function parseCSV(csv) {
            const lines = csv.split('\\n').filter(line => line.trim() !== '');
            const headers = lines[0].split(',');
            const data = lines.slice(1).map(line => {
                const values = line.split(',');
                let obj = {};
                headers.forEach((h, i) => obj[h] = values[i]);
                return obj;
            });
            return data;
        }

        // 加载CSV数据并绘制图表
        fetch('nginx_monitor_data.csv')
            .then(response => response.text())
            .then(csv => {
                const data = parseCSV(csv);
                const labels = data.map(item => item['时间']);
                const cpuData = data.map(item => item['CPU使用率(%)']);
                const memData = data.map(item => item['内存使用率(%)']);
                const connData = data.map(item => item['连接数']);
                const reqData = data.map(item => item['请求数']);

                // CPU/内存图表
                new Chart(document.getElementById('resourceChart'), {
                    type: 'line',
                    data: {
                        labels: labels,
                        datasets: [
                            { label: 'CPU使用率(%)', data: cpuData, borderColor: '#ff6384', tension: 0.3 },
                            { label: '内存使用率(%)', data: memData, borderColor: '#36a2eb', tension: 0.3 }
                        ]
                    },
                    options: { responsive: true, title: { display: true, text: 'CPU/内存使用率趋势' } }
                });

                // Nginx性能图表
                new Chart(document.getElementById('nginxChart'), {
                    type: 'line',
                    data: {
                        labels: labels,
                        datasets: [
                            { label: '活跃连接数', data: connData, borderColor: '#cc65fe', tension: 0.3 },
                            { label: '每分钟请求数', data: reqData, borderColor: '#ffce56', tension: 0.3 }
                        ]
                    },
                    options: { responsive: true, title: { display: true, text: 'Nginx连接数/请求量趋势' } }
                });
            });
    </script>
</body>
</html>
EOF
    log "INFO" "可视化报表已生成：${REPORT_FILE}"
}

# -------------------------- 主程序 --------------------------
main() {
    # 创建必要文件
    [ ! -f ${LOG_FILE} ] && touch ${LOG_FILE}
    [ ! -f ${DATA_FILE} ] && echo "时间,CPU使用率(%),内存使用率(%),连接数,请求数" > ${DATA_FILE}
    [ ! -d $(dirname ${REPORT_FILE}) ] && mkdir -p $(dirname ${REPORT_FILE})

    log "INFO" "===== 开始全维度监控 ====="
    # 基础监控
    check_nginx_process
    check_nginx_port
    check_upstream_servers
    check_nginx_config
    # 新增监控 + 生成报表
    generate_report
    log "INFO" "===== 监控完成 ====="
}

# 执行主程序
main

# 循环监控（可选，注释掉则单次执行）

# done
```

### 部署与使用说明

#### 1. 环境依赖安装

```bash
# CentOS/RHEL
yum install -y net-tools sendmail curl wget top freeutils nginx

# Ubuntu/Debian
apt-get install -y net-tools sendmail curl wget procps nginx
```

#### 2. 脚本部署步骤

```bash
# 1. 创建脚本文件并赋予执行权限
touch /usr/local/bin/nginx_full_monitor.sh
chmod +x /usr/local/bin/nginx_full_monitor.sh

# 2. 将上述脚本内容复制到文件中，修改配置项
vi /usr/local/bin/nginx_full_monitor.sh
# 重点修改：后端服务器列表、告警方式/信息、阈值配置、报表路径

# 3. 测试脚本
/usr/local/bin/nginx_full_monitor.sh

# 4. 查看监控日志和数据文件
tail -f /var/log/nginx/monitor.log
cat /var/log/nginx/monitor_data.csv
```

#### 3. 定时任务配置（核心）

```bash
# 编辑crontab，每5分钟执行一次
crontab -e

# 添加以下内容
*/5 * * * * /usr/local/bin/nginx_full_monitor.sh > /dev/null 2>&1

# 重启crond服务
systemctl restart crond
```

#### 4. 访问可视化报表

```bash
# 确保Nginx已配置静态文件访问（指向报表目录）
# 示例配置（添加到nginx.conf）：
echo '
server {
    listen 81;
    server_name localhost;
    root /var/www/html;
    index nginx_monitor.html;
}' >> /etc/nginx/conf.d/monitor.conf

# 重载Nginx
nginx -t && systemctl reload nginx

# 访问报表（浏览器打开）
# http://你的服务器IP:81/nginx_monitor.html
```

### 核心功能扩展说明

|新增监控项|检测逻辑|告警触发条件|
|---|---|---|
|CPU 使用率|基于`top`命令计算整体 CPU 使用率|超过`CPU_THRESHOLD`（默认 80%）|
|内存使用率|基于`free`命令计算内存使用率|超过`MEM_THRESHOLD`（默认 80%）|
|Nginx 连接数|统计`ESTABLISHED`状态的连接数|超过`CONN_THRESHOLD`（默认 2000）|
|每分钟请求量|分析`access.log`最近 1 分钟请求数|超过`REQ_THRESHOLD`（默认 1000）|
|可视化报表|基于 Chart.js 绘制趋势图表|自动生成 HTML 报表，支持浏览器访问|

### 总结

1. 这份增强版脚本在原有基础上，新增了**系统资源监控（CPU / 内存）** 和**Nginx 性能监控（连接数 / 请求量）**，并自动生成可视化 HTML 报表；
2. 核心配置项需修改：后端服务器列表、告警方式（邮件 / 钉钉）、阈值（CPU / 内存 / 连接数 / 请求量）、报表存储路径；
3. 定时执行后，可通过浏览器访问 `http://服务器IP:81/nginx_monitor.html` 查看实时监控图表，直观掌握 Nginx 和服务器状态；
4. 报表支持 CPU / 内存使用率、Nginx 连接数 / 请求量的趋势展示，便于排查性能瓶颈和异常波动。
