# Nginx 终极监控脚本（完整版）

```bash
#!/bin/bash
# Nginx 终极监控脚本
# 功能：进程+端口+后端服务+CPU+内存+磁盘+连接数+请求量+错误率+告警+报表

# ==================== 配置项（只需改这里）====================
NGINX_PID="/var/run/nginx.pid"
NGINX_CMD="/usr/sbin/nginx"
NGINX_CONF="/etc/nginx/nginx.conf"
MONITOR_PORT="80"
UPSTREAM_SERVERS=("127.0.0.1:8080" "127.0.0.1:8081")
LOG_FILE="/var/log/nginx/monitor.log"
DATA_FILE="/var/log/nginx/monitor_data.csv"
REPORT_FILE="/var/www/html/nginx_monitor.html"

# 告警
ALERT_TYPE="dingding"
DINGDING_WEBHOOK="https://oapi.dingtalk.com/robot/send?access_token=你的token"
TO_EMAIL="alert@xxx.com"

# 阈值
CPU_THRESHOLD=85
MEM_THRESHOLD=85
DISK_THRESHOLD=85
CONN_THRESHOLD=2000
REQ_THRESHOLD=1000
ERROR_RATE_THRESHOLD=5  # 错误率超过 5% 告警

# ==================== 工具函数 ====================
log() {
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] [$1] $2" >> $LOG_FILE
}

send_alert() {
    content="【Nginx监控告警】$2"
    curl -s -X POST $DINGDING_WEBHOOK \
        -H "Content-Type: application/json" \
        -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"$content\"}}"
    log "ALERT" "$2"
}

# ==================== 基础监控 ====================
check_nginx_process() {
    if [ ! -f $NGINX_PID ] || ! ps -p $(cat $NGINX_PID) >/dev/null 2>&1; then
        send_alert "紧急" "Nginx 进程已停止！"
        $NGINX_CMD -t && $NGINX_CMD || send_alert "严重" "Nginx 启动失败"
    fi
}

check_nginx_port() {
    if ! netstat -tulpn | grep ":$MONITOR_PORT" | grep nginx >/dev/null 2>&1; then
        send_alert "紧急" "Nginx 端口 $MONITOR_PORT 未监听"
    fi
}

check_upstream_servers() {
    unhealthy=()
    for s in "${UPSTREAM_SERVERS[@]}"; do
        ip=$(echo $s | cut -d: -f1)
        port=$(echo $s | cut -d: -f2)
        if ! timeout 2 bash -c "echo >/dev/tcp/$ip/$port" 2>/dev/null; then
            unhealthy+=("$s")
        fi
    done
    if [ ${#unhealthy[@]} -gt 0 ]; then
        send_alert "警告" "后端异常：${unhealthy[*]}"
    fi
}

# ==================== 系统监控 ====================
check_cpu() {
    cpu=$(top -bn1 | grep Cpu | awk '{print 100-$8}' | cut -d. -f1)
    [ $cpu -ge $CPU_THRESHOLD ] && send_alert "警告" "CPU 过高：$cpu%"
    echo -n "$cpu," >> $DATA_FILE
}

check_mem() {
    mem=$(free | awk '/Mem/{printf("%.0f"), $3/$2*100}')
    [ $mem -ge $MEM_THRESHOLD ] && send_alert "警告" "内存过高：$mem%"
    echo -n "$mem," >> $DATA_FILE
}

check_disk() {
    disk=$(df -h / | awk '/\//{print $5}' | tr -d %)
    [ $disk -ge $DISK_THRESHOLD ] && send_alert "警告" "磁盘使用率过高：$disk%"
    echo -n "$disk," >> $DATA_FILE
}

# ==================== Nginx 性能监控 ====================
check_connections() {
    conn=$(ss -nt | grep :$MONITOR_PORT | grep ESTAB | wc -l)
    [ $conn -ge $CONN_THRESHOLD ] && send_alert "警告" "连接数过高：$conn"
    echo -n "$conn," >> $DATA_FILE
}

check_requests() {
    start=$(date -d "1 min ago" +"%d/%b/%Y:%H:%M")
    req=$(grep "$start" /var/log/nginx/access.log 2>/dev/null | wc -l)
    [ $req -ge $REQ_THRESHOLD ] && send_alert "警告" "1分钟请求量过高：$req"
    echo -n "$req," >> $DATA_FILE
}

check_error_rate() {
    start=$(date -d "1 min ago" +"%d/%b/%Y:%H:%M")
    total=$(grep "$start" /var/log/nginx/access.log 2>/dev/null | wc -l)
    err=$(grep "$start" /var/log/nginx/access.log 2>/dev/null | awk '$9>=400' | wc -l)
    if [ $total -gt 0 ]; then
        rate=$(echo "scale=2; $err/$total*100" | bc)
        err_int=$(echo "$rate>=$ERROR_RATE_THRESHOLD" | bc)
        [ $err_int -eq 1 ] && send_alert "警告" "错误率过高：$rate%"
    else
        rate=0
    fi
    echo "$rate" >> $DATA_FILE
}

# ==================== 生成报表 ====================
gen_report() {
    [ ! -f $DATA_FILE ] && echo "时间,CPU,内存,磁盘,连接数,请求数,错误率" > $DATA_FILE
    echo -n "$(date +'%Y-%m-%d %H:%M:%S')," >> $DATA_FILE
    check_cpu
    check_mem
    check_disk
    check_connections
    check_requests
    check_error_rate
}

# ==================== 主程序 ====================
main() {
    log "INFO" "开始监控"
    check_nginx_process
    check_nginx_port
    check_upstream_servers
    gen_report
    log "INFO" "监控完成"
}

main
```
