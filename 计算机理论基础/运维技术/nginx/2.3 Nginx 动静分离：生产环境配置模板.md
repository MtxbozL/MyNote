这份配置是生产级别的，包含了动静分离核心逻辑、性能调优、安全加固、日志配置等关键内容，你只需根据自己的服务器路径和后端地址稍作修改即可使用。

```nginx
# -------------------------- 全局配置 --------------------------
user nginx;
# 自动匹配CPU核心数，充分利用多核性能
worker_processes auto;
# 错误日志级别：生产环境建议 warn/error，调试时用 debug
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

# -------------------------- 事件模块 --------------------------
events {
    # 每个worker进程的最大并发连接数
    worker_connections 10240;
    # 开启多路复用IO模型（epoll仅Linux支持，高效）
    use epoll;
    # 允许一个worker同时处理多个连接
    multi_accept on;
}

# -------------------------- HTTP核心模块 --------------------------
http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # -------------------------- 日志格式 --------------------------
    # 自定义日志格式，包含真实IP、响应时间等关键信息
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for" '
                    '$request_time $upstream_response_time';
    # 访问日志路径
    access_log /var/log/nginx/access.log main;

    # -------------------------- 性能调优 --------------------------
    # 开启高效文件传输（内核态直接传输，避免用户态拷贝）
    sendfile on;
    # 配合sendfile，合并数据包发送，减少网络开销
    tcp_nopush on;
    # 短连接快速响应，禁用Nagle算法
    tcp_nodelay on;
    # 连接超时时间（秒）
    keepalive_timeout 65;
    # 单个连接的最大请求数
    keepalive_requests 1000;

    # -------------------------- Gzip压缩（文本资源） --------------------------
    gzip on;
    # 压缩级别（1-9，级别越高压缩率越高，CPU消耗越大）
    gzip_comp_level 6;
    # 最小压缩文件大小（小于此值不压缩）
    gzip_min_length 256;
    # 压缩的文件类型
    gzip_types text/plain text/css application/json application/javascript 
               text/xml application/xml application/xml+rss text/javascript
               application/vnd.ms-fontobject application/x-font-ttf font/opentype
               image/svg+xml image/x-icon;
    # 对代理请求也启用压缩
    gzip_proxied any;
    # 兼容老版本IE
    gzip_vary on;

    # -------------------------- 后端服务集群（负载均衡） --------------------------
    upstream backend_servers {
        # 后端应用服务器地址，weight=权重（数值越大处理请求越多）
        server 127.0.0.1:8080 weight=5;
        # 可添加多个后端节点，示例：
        # server 127.0.0.1:8081 weight=3;
        # server 10.0.0.2:8080 backup;  # 备份节点（主节点不可用时启用）
        
        # 负载均衡策略（默认轮询，可选：ip_hash/least_conn）
        # ip_hash;  # 按客户端IP哈希，保证同一客户端请求到同一后端
        # least_conn;  # 最少连接数优先
    }

    # -------------------------- 虚拟主机配置（核心） --------------------------
    server {
        # 监听端口（HTTP）
        listen 80;
        # 你的域名（多个域名用空格分隔）
        server_name your-domain.com www.your-domain.com;
        
        # 强制跳转HTTPS（如果配置了SSL，建议开启）
        # return 301 https://$host$request_uri;

        # -------------------------- 静态资源配置 --------------------------
        # 1. 网站图标（精确匹配，优先级最高）
        location = /favicon.ico {
            root /var/www/static/icons;
            expires 30d;  # 浏览器缓存30天
            access_log off;  # 关闭该路径的访问日志
        }

        # 2. 静态资源目录（前缀匹配，优先级高于正则）
        location ^~ /static/ {
            root /var/www;  # 实际路径：/var/www/static/xxx
            expires 7d;     # 缓存7天
            add_header Cache-Control "public, max-age=604800";
            # 限制静态资源访问速度（可选，防止大文件占用带宽）
            # limit_rate 1024k;
        }

        # 3. 媒体文件（图片/视频/音频，正则匹配）
        location ~* \.(jpg|jpeg|png|gif|webp|mp4|avi|mp3|wav)$ {
            root /var/www/media;
            expires 30d;
            add_header Cache-Control "public, max-age=2592000";
            access_log off;
        }

        # 4. 样式/脚本文件（正则匹配）
        location ~* \.(css|js|json|svg|woff|woff2|ttf|eot)$ {
            root /var/www/assets;
            expires 7d;
            add_header Cache-Control "public, max-age=604800";
            access_log off;
        }

        # -------------------------- 动态请求配置 --------------------------
        location / {
            # 转发到后端服务集群
            proxy_pass http://backend_servers;
            
            # -------------------------- 反向代理核心头信息 --------------------------
            # 传递客户端真实IP
            proxy_set_header X-Real-IP $remote_addr;
            # 传递经过的代理服务器IP（便于后端溯源）
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            # 传递原始请求的协议（HTTP/HTTPS）
            proxy_set_header X-Forwarded-Proto $scheme;
            # 传递请求的域名
            proxy_set_header Host $host;
            
            # -------------------------- 反向代理超时配置 --------------------------
            # 连接后端超时时间
            proxy_connect_timeout 30s;
            # 读取后端响应超时时间
            proxy_read_timeout 60s;
            # 发送请求到后端超时时间
            proxy_send_timeout 30s;
            
            # -------------------------- 缓冲区配置 --------------------------
            proxy_buffering on;
            # 响应头缓冲区大小（建议与系统页大小一致，一般4k/8k）
            proxy_buffer_size 4k;
            # 响应体缓冲区（数量+大小）
            proxy_buffers 8 4k;
            # 大响应的临时缓冲区
            proxy_busy_buffers_size 8k;
        }

        # -------------------------- 安全加固 --------------------------
        # 禁止访问敏感文件
        location ~* \.(conf|bak|sql|log|sh)$ {
            deny all;
        }

        # 限制请求方法（仅允许GET/POST/HEAD）
        if ($request_method !~ ^(GET|POST|HEAD)$) {
            return 403;
        }

        # -------------------------- 错误页面配置 --------------------------
        error_page 404 /404.html;
        error_page 500 502 503 504 /50x.html;
        location = /50x.html {
            root /var/www/static/error;
        }
    }

    # -------------------------- HTTPS配置（可选，生产建议开启） --------------------------
    # server {
    #     listen 443 ssl http2;
    #     server_name your-domain.com www.your-domain.com;

    #     # SSL证书路径
    #     ssl_certificate /etc/nginx/ssl/your-domain.crt;
    #     ssl_certificate_key /etc/nginx/ssl/your-domain.key;

    #     # SSL优化配置
    #     ssl_protocols TLSv1.2 TLSv1.3;
    #     ssl_prefer_server_ciphers on;
    #     ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES256-GCM-SHA384;
    #     ssl_session_timeout 10m;
    #     ssl_session_cache shared:SSL:10m;
    #     ssl_session_tickets off;

    #     # 静态/动态请求配置（同HTTP部分，可直接复用）
    #     location = /favicon.ico { ... }
    #     location ^~ /static/ { ... }
    #     location / { ... }
    # }
}
```

### 部署前准备步骤

1. **创建目录结构**（根据配置中的路径）：

```bash
    # 创建静态资源目录
    mkdir -p /var/www/{static/icons,media,assets,static/error}
    # 创建日志目录（如果不存在）
    mkdir -p /var/log/nginx
    # 创建SSL证书目录（如果启用HTTPS）
    mkdir -p /etc/nginx/ssl
    ```

2. **修改关键配置项**（必须根据你的实际环境调整）：
    
    - `server_name`：替换为你的实际域名（如 `example.com`）。
    - `root` 路径：确保配置中的 `/var/www/xxx` 路径与你服务器上的静态资源路径一致。
    - `upstream backend_servers`：替换为你的后端应用服务器地址和端口（如 Tomcat/Java/Node.js 服务）。
    - SSL 证书路径：如果启用 HTTPS，替换为你的证书和私钥路径。
    
3. **配置验证与重载**：

```bash
    # 验证配置语法是否正确
    nginx -t
    # 重载配置（不中断服务）
    systemctl reload nginx
    # 启动/重启Nginx
    systemctl start nginx  # 启动
    systemctl restart nginx  # 重启
    ```

### 总结

1. 这份模板是**生产级**的，包含了动静分离核心逻辑、性能调优（sendfile/gzip/ 连接数）、安全加固（禁用敏感文件访问 / 限制请求方法）、反向代理关键配置（真实 IP 传递 / 超时控制）。
2. 核心修改点：`server_name`（域名）、`root`（静态资源路径）、`upstream`（后端服务地址），这三项必须根据你的实际环境调整。
3. 生产环境建议启用 HTTPS（取消模板中 HTTPS 部分的注释，配置证书），并定期检查 Nginx 日志（`/var/log/nginx/`）监控运行状态。
