这份配置在生产级动静分离基础上，新增了 WebSocket 实时通信代理、子请求接口聚合、健康检查等高级特性，注释详尽且兼顾性能与稳定性。

```nginx
# -------------------------- 全局配置 --------------------------
user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

# -------------------------- 事件模块 --------------------------
events {
    worker_connections 10240;
    use epoll;
    multi_accept on;
}

# -------------------------- HTTP核心模块 --------------------------
http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # 日志格式（新增upstream_addr记录后端地址）
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for" '
                    '$request_time $upstream_response_time $upstream_addr';
    access_log /var/log/nginx/access.log main;

    # -------------------------- 基础性能调优 --------------------------
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    keepalive_requests 1000;

    # Gzip压缩（保留原有配置）
    gzip on;
    gzip_comp_level 6;
    gzip_min_length 256;
    gzip_types text/plain text/css application/json application/javascript 
               text/xml application/xml application/xml+rss text/javascript
               application/vnd.ms-fontobject application/x-font-ttf font/opentype
               image/svg+xml image/x-icon;
    gzip_proxied any;
    gzip_vary on;

    # -------------------------- 后端服务集群（含健康检查） --------------------------
    upstream backend_servers {
        # 基础后端应用节点
        server 127.0.0.1:8080 weight=5 max_fails=3 fail_timeout=30s;
        server 127.0.0.1:8081 weight=3 max_fails=3 fail_timeout=30s;
        
        # WebSocket专用后端（可单独部署，也可复用应用节点）
        server 127.0.0.1:8082 weight=2 max_fails=3 fail_timeout=30s;

        # 负载均衡策略
        least_conn;  # 最少连接数优先（适合长连接/WebSocket场景）
        
        # 开启被动健康检查（Nginx原生支持，无需额外模块）
        # max_fails：失败次数；fail_timeout：失败后隔离时间
    }

    # 子请求专用后端（如微服务接口、数据聚合服务）
    upstream subrequest_servers {
        server 127.0.0.1:9000;  # 接口聚合服务
        server 127.0.0.1:9001;  # 用户信息服务
    }

    # -------------------------- 虚拟主机核心配置 --------------------------
    server {
        listen 80;
        server_name your-domain.com www.your-domain.com;
        # 生产环境建议强制HTTPS
        # return 301 https://$host$request_uri;

        # -------------------------- 1. 静态资源配置（复用原有） --------------------------
        location = /favicon.ico {
            root /var/www/static/icons;
            expires 30d;
            access_log off;
        }

        location ^~ /static/ {
            root /var/www;
            expires 7d;
            add_header Cache-Control "public, max-age=604800";
            access_log off;
        }

        location ~* \.(jpg|jpeg|png|gif|webp|mp4|css|js)$ {
            root /var/www/media;
            expires 30d;
            access_log off;
        }

        # -------------------------- 2. WebSocket代理配置（核心新增） --------------------------
        # 匹配WebSocket请求路径（如 /ws/chat、/ws/push）
        location /ws/ {
            # 转发到WebSocket后端
            proxy_pass http://backend_servers;
            
            # WebSocket核心配置（必须）
            proxy_http_version 1.1;  # 强制使用HTTP/1.1（WebSocket基于此）
            proxy_set_header Upgrade $http_upgrade;  # 升级协议头
            proxy_set_header Connection "upgrade";  # 连接类型：升级
            
            # 传递客户端信息
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            
            # WebSocket长连接超时配置（关键：避免连接被断开）
            proxy_read_timeout 86400s;  # 长连接超时（24小时，根据业务调整）
            proxy_send_timeout 86400s;
            proxy_connect_timeout 30s;
            
            # 禁用缓冲区（实时通信无需缓冲）
            proxy_buffering off;
        }

        # -------------------------- 3. 子请求（Subrequest）配置（核心新增） --------------------------
        # 场景：前端请求 /api/user 时，Nginx自动聚合多个后端接口数据返回
        location /api/user {
            # 启用子请求响应体聚合（需要ngx_http_subrequest_output_filter_module）
            subrequest_output_buffer_size 4k;
            
            # 默认响应类型
            default_type application/json;
            
            # 子请求逻辑：通过 rewrite + echo 实现（需安装 ngx_http_echo_module 模块）
            # 步骤1：发起子请求获取用户基本信息
            rewrite ^ /_internal/user/info break;
            proxy_pass http://subrequest_servers;
            
            # 步骤2：发起子请求获取用户权限（通过 echo_subrequest 实现并行子请求）
            # 注：需先安装 echo 模块（nginx -V 查看是否包含 --add-module=echo-nginx-module）
            echo_subrequest GET /_internal/user/permission @permission;
            
            # 步骤3：合并响应（示例：简单拼接，复杂场景可通过Lua脚本处理）
            echo '{"user_info": $echo_response_body, "permission": $echo_subrequest_response_body}';
        }

        # 子请求内部路径（对外不可访问）
        location /_internal/ {
            internal;  # 标记为内部路径，仅允许Nginx内部子请求访问
            proxy_pass http://subrequest_servers;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_connect_timeout 10s;
            proxy_read_timeout 10s;
        }

        # -------------------------- 4. 常规动态请求配置 --------------------------
        location / {
            proxy_pass http://backend_servers;
            
            # 反向代理基础配置
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            
            # 超时与缓冲区配置
            proxy_connect_timeout 30s;
            proxy_read_timeout 60s;
            proxy_send_timeout 30s;
            proxy_buffer_size 4k;
            proxy_buffers 8 4k;
            proxy_busy_buffers_size 8k;
        }

        # -------------------------- 5. 安全与错误配置 --------------------------
        # 禁止访问敏感文件
        location ~* \.(conf|bak|sql|log|sh)$ {
            deny all;
        }

        # 限制请求方法
        if ($request_method !~ ^(GET|POST|HEAD|PUT|DELETE|OPTIONS)$) {
            return 403;
        }

        # 错误页面
        error_page 404 /404.html;
        error_page 500 502 503 504 /50x.html;
        location = /50x.html {
            root /var/www/static/error;
        }
    }

    # -------------------------- HTTPS配置（生产建议开启） --------------------------
    server {
        listen 443 ssl http2;
        server_name your-domain.com www.your-domain.com;

        # SSL证书配置
        ssl_certificate /etc/nginx/ssl/your-domain.crt;
        ssl_certificate_key /etc/nginx/ssl/your-domain.key;

        # SSL优化
        ssl_protocols TLSv1.2 TLSv1.3;
        ssl_prefer_server_ciphers on;
        ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512;
        ssl_session_timeout 10m;
        ssl_session_cache shared:SSL:10m;
        ssl_session_tickets off;

        # 复用上述所有配置（静态/WS/子请求/动态）
        location = /favicon.ico { ... }
        location ^~ /static/ { ... }
        location /ws/ { ... }
        location /api/user { ... }
        location /_internal/ { ... }
        location / { ... }
    }
}
```

### 关键依赖与部署说明

#### 1. 模块依赖（子请求核心）

Nginx 原生子请求能力有限，如需实现**并行子请求、响应体合并**，需安装扩展模块：

```bash
# 1. 检查已安装模块（确认是否包含echo模块）
nginx -V

# 2. 若未安装，重新编译Nginx（以CentOS为例）
# 下载echo模块
wget https://github.com/openresty/echo-nginx-module/archive/refs/tags/v0.61.tar.gz
tar -zxvf v0.61.tar.gz

# 重新编译（保留原有配置，添加echo模块）
./configure --prefix=/usr/local/nginx --add-module=./echo-nginx-module-0.61
make && make install
```

#### 2. 核心配置调整点

|配置项|调整说明|
|---|---|
|`server_name`|替换为你的实际域名（如 `example.com`）|
|`upstream` 节点|替换为你的后端服务地址（WebSocket / 子请求 / 常规应用）|
|`/ws/` 路径|根据业务调整 WebSocket 请求路径（如 `/ws/chat` 改为 `/realtime/push`）|
|`/api/user` 子请求|根据接口聚合需求，修改子请求路径和后端地址|
|`proxy_read_timeout`|WebSocket 场景建议设置为长超时（如 86400s），避免长连接被 Nginx 主动断开|

#### 3. 验证与启动

```bash
# 验证配置语法
nginx -t

# 重载配置（不中断服务）
systemctl reload nginx

# 查看WebSocket连接状态（验证是否生效）
netstat -an | grep 8082  # 检查WebSocket后端端口连接数
```

### 总结

1. **WebSocket 核心**：必须配置 `proxy_http_version 1.1`、`Upgrade`/`Connection` 头、长超时时间，禁用缓冲区，确保实时通信不中断。
2. **子请求核心**：通过 `internal` 标记内部路径，结合 `echo_subrequest` 实现多接口聚合，适合微服务接口统一收口场景。
3. **生产注意事项**：子请求建议配合 `Lua` 脚本（OpenResty）实现更灵活的响应合并；WebSocket 需监控连接数，避免后端过载；HTTPS 必须开启，WebSocket 对应 `wss://` 协议（兼容浏览器安全策略）。
